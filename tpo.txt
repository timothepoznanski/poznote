Install :

cd /root/poznote/poznote/ && docker compose -f docker-compose.yml -f docker-compose-reverse-proxy.yml up -d

Update :


cd /root/poznote/poznote/ && git pull origin main && docker compose down && docker compose -f docker-compose.yml -f docker-compose-reverse-proxy.yml up -d --build

Demo install :

docker network create npm-poznote-demo-webserver-net
docker network create poznote-demo_internal-net
